2.Remove duplicates from sorted array

class Solution {
public:
    int removeDuplicates(vector<int>& nums) 
    {

        auto it = unique(nums.begin(),nums.end());
        nums.erase(it, nums.end());
       int k = 0;
        for( auto i =0;i<nums.size();i++){
            k++;
        }

       return k;
    }

};
